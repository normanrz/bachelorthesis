\section{General}

\begin{lstlisting}[language=restinterface]
GET /resources
# returns the 50 first resources

GET /resources?limit=100
# returns the 100 first resources

GET /resources?offset=30&limit=10
# returns 10 resources from the 30th

GET /resources/:id
# returns all details of a specific resource


PUT /resources/:id
# replaces this item 
=> 200 OK
or
=> 400 Bad Request
   {
     errors: []
   }


PATCH /resources/:id
# incrementally updates this item
=> 200 OK
or
=> 400 Bad Request
   {
     errors: []
   } 


POST /resources
# creates a new item
=> 200 OK
or
=> 400 Bad Request
   {
     errors: []
   }


DELETE /resources/:id
# deletes this item
=> 200 OK
or
=> 400 Bad Request
   {
     errors: []
   }

\end{lstlisting}
